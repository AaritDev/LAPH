def parse_dice_input(input_str):
    dice_list = []
    for die in input_str.split(','):
        die = die.strip()
        if not die:

        parts = die.split('d')
        if len(parts) != 2:
            raise ValueError(f"Invalid die format: {die}")
        count, sides = map(int, parts)

            raise ValueError(f"Invalid count or sides for die: {die}")
        dice_list.append((count, sides))
    return dice_list
